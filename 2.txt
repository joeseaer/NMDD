## 🛠️ 性格分析档案数据库设计

```sql
-- 人物档案表
CREATE TABLE people_profiles (
    id UUID PRIMARY KEY,
    user_id UUID NOT NULL,
    name VARCHAR(100) NOT NULL,
    identity VARCHAR(200),          -- 身份/职位
    field VARCHAR(200),             -- 领域/行业
    contact_info JSONB,             -- 联系方式 (微信/邮箱/电话)
    met_date DATE,                  -- 认识时间
    met_context TEXT,               -- 认识场景
    tags TEXT[],                    -- 标签数组
    relationship_strength INT,      -- 关系强度 0-100
    disc_type VARCHAR(10),          -- DISC类型
    mbti_type VARCHAR(10),          -- MBTI类型
    ai_analysis TEXT,               -- AI性格分析
    interaction_tips TEXT,          -- 相处建议
    created_at TIMESTAMP DEFAULT NOW(),
    updated_at TIMESTAMP DEFAULT NOW()
);

-- 互动记录表
CREATE TABLE interaction_logs (
    id UUID PRIMARY KEY,
    person_id UUID REFERENCES people_profiles(id),
    event_date DATE NOT NULL,
    event_context TEXT,             -- 事件描述
    my_behavior TEXT,               -- 我的表现
    their_reaction TEXT,            -- 对方反应
    relationship_change INT,        -- 关系变化
    ai_analysis TEXT,               -- AI分析
    created_at TIMESTAMP DEFAULT NOW()
);

-- 性格分析版本表
CREATE TABLE personality_versions (
    id UUID PRIMARY KEY,
    person_id UUID REFERENCES people_profiles(id),
    version VARCHAR(20) NOT NULL,   -- V1.0, V2.0...
    disc_type VARCHAR(10),
    mbti_type VARCHAR(10),
    analysis_notes TEXT,
    created_at TIMESTAMP DEFAULT NOW()
);
```
