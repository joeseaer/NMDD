## 🚀 部署流程（一键部署脚本）
```bash
#!/bin/bash
# 阿里云轻量服务器一键部署脚本

# 1. 系统更新
apt update && apt upgrade -y

# 2. 安装Node.js
curl -fsSL https://deb.nodesource.com/setup_20.x | bash -
apt install -y nodejs

# 3. 安装PostgreSQL
apt install -y postgresql postgresql-contrib

# 4. 安装Nginx
apt install -y nginx

# 5. 安装PM2
npm install -g pm2

# 6. 克隆代码
git clone https://github.com/your-repo/entrepreneur-simulator.git
cd entrepreneur-simulator

# 7. 安装依赖
npm install

# 8. 配置环境变量
cp .env.example .env
# 编辑 .env 填入API密钥等

# 9. 初始化数据库
npm run db:migrate

# 10. 启动应用
pm2 start ecosystem.config.js
pm2 save
pm2 startup

# 11. 配置Nginx
cp nginx.conf /etc/nginx/sites-available/entrepreneur-simulator
ln -s /etc/nginx/sites-available/entrepreneur-simulator /etc/nginx/sites-enabled/
nginx -t && systemctl restart nginx

# 12. 配置SSL (免费Let's Encrypt)
apt install -y certbot python3-certbot-nginx
certbot --nginx -d your-domain.com

echo "✅ 部署完成！"

```
